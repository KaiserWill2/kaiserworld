country_event = { #LBA unites Libya
	id = libyan_unification.1
	title = libyan_unification.1.t
	desc = libyan_unification.1.d
	picture = unified_libya
	
	fire_only_once = yes
	
	trigger = {
        tag = LBA
        LBA = {
            has_idea = idea_libyan_civil_war
        }
        OR = {
            NOT = {
                country_exists = GNS
            }
            GNS = {
                is_puppet_of = LBA
            }
        }
        OR = {
            NOT = {
                country_exists = FAL
            }
            FAL = {
                is_puppet_of = LBA
            }
        }
        OR = {
            NOT = {
                country_exists = ISL
            }
            ISL = {
                is_puppet_of = LBA
            }
        }
        OR = {
            NOT = {
                country_exists = RAL
            }
            RAL = {
                is_puppet_of = LBA
            }
        }
        OR = {
            NOT = {
                country_exists = ACS
            }
            ACS = {
                is_puppet_of = LBA
            }
        }
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = { 
		name = libyan_unification.1.o1
		ai_chance = {
			factor = 100
		}
        LBA = {
            annex_country = {
                target = GNS
                transfer_troops = no
            }
            annex_country = {
                target = FAL
                transfer_troops = no
            }
            annex_country = {
                target = ISL
                transfer_troops = no
            }
            annex_country = {
                target = RAL
                transfer_troops = no
            }
            annex_country = {
                target = ACS
                transfer_troops = no
            }
            remove_ideas = idea_libyan_civil_war
        }
        GNS = {
            remove_ideas = idea_libyan_civil_war
        }
        FAL = {
            remove_ideas = idea_libyan_civil_war
        }
        ISL = {
            remove_ideas = idea_libyan_civil_war
        }
        RAL = {
            remove_ideas = idea_libyan_civil_war
        }
        ACS = {
            remove_ideas = idea_libyan_civil_war
        }
		set_cosmetic_tag = LBA_United
		add_stability = 0.15
		every_state = {
		    limit = {
			    is_core_of = GNS
			}
			remove_core_of = GNS
		}
        every_state = {
		    limit = {
			    is_core_of = FAL
			}
			remove_core_of = FAL
		}
        every_state = {
		    limit = {
			    is_core_of = ISL
			}
			remove_core_of = ISL
		}
        every_state = {
		    limit = {
			    is_core_of = RAL
			}
			remove_core_of = RAL
		}
        every_state = {
		    limit = {
			    is_core_of = ACS
			}
			remove_core_of = ACS
		}
        hidden_effect = {
            every_country = {
                limit = {
                    NOT = {
                        tag = LBA
                    }
                }
                news_event = {
                    hours = 4
                    id = news.238
                }
            }
        }
	}
}

add_namespace = election

#Election event
country_event = {
	id = election.1
	title = election.1.t
	desc = election.1.d
	picture = generic_election_event
	
	is_triggered_only = yes
	
	
	option = {	#Current administration
		name = election.1.o12
		ai_chance = {
			factor = 12
		}
		trigger = {
			OR = {
				AND = {
				    has_government = moderate_socialist
					moderate_socialist > 0.2
				}
				AND = {
				    has_government = new_democrat
					new_democrat > 0.2
				}
				AND = {
				    has_government = liberal
					liberal > 0.2
				}
				AND = {
				    has_government = centrist
					centrist > 0.2
				}
				AND = {
				    has_government = national_conservative
					national_conservative > 0.2
				}
				AND = {
					moderate_socialist < 0.2
					new_democrat < 0.2
					liberal < 0.2
					centrist < 0.2
					national_conservative < 0.2
				}
			}
		}
		
		
	}
	
	option = {	#National Conservatives
		name = election.1.o1
		ai_chance = {
			factor = 3
		}
		trigger = {
			NOT = { has_government = national_conservative }
			national_conservative > 0.19
		}
		
		set_politics = {
			ruling_party = national_conservative
			elections_allowed = no
		}
	}
	
	option = {	#Centrists
		name = election.1.o2
		ai_chance = {
			factor = 3
		}
		trigger = {
			NOT = { has_government = centrist }
			centrist > 0.19
		}
		
		set_politics = {
			ruling_party = centrist
			elections_allowed = no
		}
	}
	
	option = {	#Liberals
		name = election.1.o3
		ai_chance = {
			factor = 3
		}
		trigger = {
			NOT = { has_government = liberal }
			liberal > 0.19
		}
		
		set_politics = {
			ruling_party = liberal
			elections_allowed = no
		}
	}
	
	option = {	#New Democrats
		name = election.1.o4
		ai_chance = {
			factor = 3
		}
		trigger = {
			NOT = { has_government = new_democrat }
			new_democrat > 0.19
		}
		
		set_politics = {
			ruling_party = new_democrat
			elections_allowed = no
		}
	}
	
	option = {	#Moderate Socialists
		name = election.1.o5
		ai_chance = {
			factor = 3
		}
		trigger = {
			NOT = { has_government = moderate_socialist }
			moderate_socialist > 0.19
		}
		
		set_politics = {
			ruling_party = moderate_socialist
			elections_allowed = no
		}
	}
	
}
